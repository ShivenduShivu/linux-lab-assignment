lsblk                       # available storage/block devices
df -Th                      # mounted filesystems + type
mount | grep '^/dev'        # real device mounts
df -h                       # disk usage statistics
du -sh ~                    # biggest space consumer (home)
df -i                       # inode utilization
free -h                     # memory context (supporting)
